#version 330 core
//in vec4 v_colour; 
in vec4 ambientLight;
in vec4 diffuseLight;
in vec4 specularLight;

// texture if maped
uniform sampler2D ourTexture;

// screen effect
uniform int u_effect;

uniform bool u_textured = false;
in vec2 TexCoord;

// result of shader
out vec4 result;

void main()  
{
	if (u_textured)
		result = texture(ourTexture, TexCoord) + (ambientLight + specularLight); 
	else
		//result = texture(ourTexture, vec2(0,0));
		result = ambientLight + diffuseLight + specularLight;
}
